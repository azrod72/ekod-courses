# Tableaux

```js
table = ["zebi",5,true] // Ajoute des éléments dans le tableau séparés par une virgule
```

```js
table.push("Hello world!") // Ajoute un élément dans le tableau (individuellement)
table.push(5) // ""
table.push(true) // ""
```

```js
table = ["zebi",365]
table.push("Hello world!")
table.push(5)
table.push(true)
console.log(table) // Affiche le tableau dans la console
console.log(table[0]) // Permet d'afficher un élément précis du tableau (ici "zebi")
```

```js
table = ["zebi",365]
table.push("Hello world!")
table.push(5)
table.push(true)
console.log(table)
console.log(table[3])
table[3] = "shesh" // Change une valeur dans le tableau
console.log(table[3])
```
